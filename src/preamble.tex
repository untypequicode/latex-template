% Hide numbers for all sectioning levels in the document body while
% keeping the internal counters so the sections / subsections still
% appear with numbers in the Table of Contents and references.
% We remove the printed label but preserve the counter for:
%   \section, \subsection, \subsubsection, \paragraph, \subparagraph
\titleformat{\section}[hang]{\normalfont\Large\bfseries}{}{0pt}{}
\titleformat{\subsection}[hang]{\normalfont\large\bfseries}{}{0pt}{}
\titleformat{\subsubsection}[hang]{\normalfont\normalsize\bfseries}{}{0pt}{}
\titleformat{\paragraph}[runin]{\normalfont\normalsize\bfseries}{}{0pt}{}
\titleformat{\subparagraph}[runin]{\normalfont\normalsize\bfseries}{}{0pt}{}

% Load the central env package if present. env.sty is expected to only
% declare variables (no side effects). Do NOT fall back to \input{env.tex}
% because that could execute arbitrary code; keep behaviour deterministic.
\IfFileExists{env.sty}{%
  \usepackage{env}%
}{%
  % env.sty not present; proceed without loading a project env package
}

% Apply graphic search paths:
% - If env.sty provided \envGraphicPaths, use it.
% - Otherwise fall back to the project's default image folder.
\makeatletter
\@ifundefined{envGraphicPaths}{%
  \graphicspath{{img/}}%
}{%
  \graphicspath\envGraphicPaths%
}
\makeatother
